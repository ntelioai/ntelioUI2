import { Widget } from '../core/Widget.js'
import { I18n } from '../utils/I18n.js'

/**
 * @typedef {'text'|'password'|'number'|'email'|'textarea'|'checkbox'|'rating'|'select'|'dropDown'|'range'|'file'|'multiDropdown'|'multiValue'|'calendar'|'date'} FieldType
 */

/**
 * @typedef {Object} FieldOption
 * @property {string} value  - Option value submitted with the form
 * @property {string} label  - Display label shown in dropdowns/tags
 */

/**
 * @typedef {Object} FieldDef
 * @property {string}            name             - Unique field identifier (used as HTML id and data key)
 * @property {FieldType}         type             - Input control type
 * @property {string}            [label]          - I18n key resolved via `I18n.t()`. Falls back to `name`
 * @property {*}                 [value]          - Initial value. Arrays for multi-types, boolean for checkbox
 * @property {*}                 [defaultValue]   - Fallback if `value` is not set
 * @property {string}            [placeholder]    - Input placeholder (defaults to resolved label)
 * @property {boolean}           [required]       - Marks the field as required for validation
 * @property {boolean}           [disabled]       - Renders the field as disabled
 * @property {boolean}           [checked]        - Initial checked state (checkbox only)
 * @property {boolean}           [multiple]       - Allow multiple selections (select only)
 * @property {number}            [size]           - Visible option count (select only)
 * @property {number}            [rows]           - Textarea row count (default 3)
 * @property {number}            [min]            - Minimum for number/range (default 0)
 * @property {number}            [max]            - Maximum for number/range (default 100)
 * @property {number}            [step]           - Step increment for number/range (default 1)
 * @property {string}            [pattern]        - Regex pattern string for validation
 * @property {string}            [patternMessage] - Custom message when pattern fails
 * @property {number}            [minLength]      - Minimum character length
 * @property {number}            [maxLength]      - Maximum character length
 * @property {Array<string|FieldOption>} [options] - Choices for select/dropDown/multiDropdown
 * @property {Array<string|FieldOption>} [enum]    - Alias for options
 * @property {string}            [dateFormat]     - jQuery datepicker format (calendar/date, default 'mm/dd/yy')
 * @property {string}            [minDate]        - Earliest selectable date (calendar/date)
 * @property {string}            [maxDate]        - Latest selectable date (calendar/date)
 * @property {string}            [yearRange]      - Year dropdown range (calendar/date, default '-100:+10')
 * @property {function(*, FieldDef, Object):string|undefined} [customValidator] - Returns error string or undefined
 */

/**
 * @typedef {Object} AutoformConfig
 * @property {string}      [title='']               - Header text shown above the form
 * @property {FieldDef[]}  fields                    - Array of field definitions
 * @property {string}      [submitLabel='Submit']    - Label for the submit button
 * @property {string}      [secondaryLabel]          - Label for the secondary button (auto-derived from `secondaryAction`)
 * @property {'reset'|'skip'} [secondaryAction='reset'] - Secondary button behaviour
 * @property {boolean}     [showSecondary=true]      - Whether to show the secondary button
 */

/**
 * @typedef {Object} SubmitPayload
 * @property {Object<string,*>} data       - Field name → value map
 * @property {string}           stringified - HTML-formatted summary of all fields
 */

/**
 * @typedef {Object} ValidationError
 * @property {string}   field   - Field name
 * @property {string}   label   - Resolved display label
 * @property {string[]} errors  - List of error messages for this field
 */

/**
 * @typedef {function(Autoform, FieldDef):void} FieldInitializer
 * Post-render setup function for complex field types.

/**
 * Declarative, config-driven form widget.
 *
 * Renders a complete form from a `fields` array, handles validation,
 * provides configurable submit/reset/skip behaviour, and caches data
 * before DOM teardown so callers always get consistent values.
 *
 * Supports 15 field types out of the box, including complex multi-value
 * selectors, date pickers, and star ratings.
 *
 * @extends Widget
 *
 * @fires Autoform#submit  - `{data, stringified}` after successful validation
 * @fires Autoform#skip    - `{stringified}` when secondary action is 'skip'
 * @fires Autoform#reset   - (no payload) after all fields are cleared
 *
 * @example
 * const form = new Autoform({
 *     title: 'Contact Us',
 *     fields: [
 *         { name: 'name',  type: 'text',  label: 'name',  required: true },
 *         { name: 'email', type: 'email', label: 'email', pattern: '^[^@]+@[^@]+\\.[^@]+$' },
 *         { name: 'message', type: 'textarea', label: 'message', rows: 4 }
 *     ]
 * })
 * form.on('submit', ({ data, stringified }) => console.log(data))
 * form.init().then(() => form.appendTo($('#container')))
 * @category Forms
 */

    // ─── Field Initializers (post-render setup for complex types) ───

    /**
     * Post-render initializers for complex field types.
     * Each entry is a {@link FieldInitializer} keyed by {@link FieldType}.
     * Called automatically by {@link _initializeFields} after HTML is appended.
     * @type {Object<FieldType, FieldInitializer>}
     */
    static fieldInitializers = {
        range: (form, field) => {
            const name = field.name
            const rangeInput = form.find(`#${name}`)
            const rangeLabel = form.find(`#${name}-label`)
            rangeLabel.text(rangeInput.val())
            rangeInput.on('input', () => {
                rangeLabel.text(rangeInput.val())
            })
        },

        multiDropdown: (form, field) => {
            const name = field.name
            const select = form.find(`#${name}-select`)
            const addButton = form.find(`#${name}-add`)
            const hiddenInput = form.find(`#${name}`)
            const tagsContainer = form.find(`#${name}-tags`)

            const allOptions = Array.isArray(field.options) ? field.options
                : Array.isArray(field.enum) ? field.enum : []

            function getValues() {
                try {
                    const parsed = JSON.parse(hiddenInput.val())
                    return Array.isArray(parsed) ? parsed : []
                } catch { return [] }
            }

            function setValues(values) {
                hiddenInput.val(JSON.stringify(values))
                renderTags(values)
            }

            function getLabelFromValue(value) {
                const match = allOptions.find(opt =>
                    (typeof opt === 'string') ? opt === value : opt.value === value
                )
                if (typeof match === 'string') return match
                return match?.label || value
            }

            function renderTags(values) {
                tagsContainer.html('')
                values.forEach((val, idx) => {
                    const label = getLabelFromValue(val)
                    tagsContainer.append(`
                        <span class="tag">${label}
                            <button class="remove-tag" data-index="${idx}">&times;</button>
                        </span>
                    `)
                })
                updateSelectOptions(values)
            }

            function updateSelectOptions(values) {
                select.html('<option value="" disabled selected>Select an option</option>')
                allOptions.forEach(option => {
                    const val = typeof option === 'string' ? option : option.value
                    const label = typeof option === 'string' ? option : option.label
                    if (!values.includes(val)) {
                        select.append(`<option value="${val}">${label}</option>`)
                    }
                })
            }

            renderTags(getValues())

            addButton.on('click', () => {
                const val = select.val()
                if (!val) return
                const values = getValues()
                if (!values.includes(val)) {
                    values.push(val)
                    setValues(values)
                    select.prop('selectedIndex', 0)
                }
            })

            tagsContainer.on('click', '.remove-tag', function () {
                const index = $(this).data('index')
                const values = getValues()
                values.splice(index, 1)
                setValues(values)
            })
        },

        multiValue: (form, field) => {
            const name = field.name
            const input = form.find(`#${name}-input`)
            const addButton = form.find(`#${name}-add`)
            const hiddenInput = form.find(`#${name}`)
            const tagsContainer = form.find(`#${name}-tags`)

            function getValues() {
                try {
                    const parsed = JSON.parse(hiddenInput.val())
                    return Array.isArray(parsed) ? parsed : []
                } catch { return [] }
            }

            function setValues(values) {
                hiddenInput.val(JSON.stringify(values))
                renderTags(values)
            }

            function renderTags(values) {
                tagsContainer.html('')
                values.forEach((val, idx) => {
                    tagsContainer.append(`
                        <span class="tag">${val}
                            <button class="remove-tag" data-index="${idx}">&times;</button>
                        </span>
                    `)
                })
            }

            renderTags(getValues())
            addButton.prop('disabled', true)

            addButton.on('click', () => {
                const val = input.val().trim()
                if (!val) return
                const values = getValues()
                if (!values.includes(val)) {
                    values.push(val)
                    setValues(values)
                    input.val('')
                    addButton.prop('disabled', true)
                }
            })

            input.on('input', () => {
                const val = input.val().trim()
                const values = getValues()
                addButton.prop('disabled', !val || values.includes(val))
            })

            input.on('keypress', (e) => {
                if (e.key === 'Enter') {
                    e.preventDefault()
                    addButton.trigger('click')
                }
            })

            tagsContainer.on('click', '.remove-tag', function () {
                const index = $(this).data('index')
                const values = getValues()
                values.splice(index, 1)
                setValues(values)
            })
        },

        calendar: (form, field) => {
            const name = field.name
            const input = form.find(`#${name}`)
            const trigger = form.find(`#${name}-trigger`)
            const dateFormat = field.dateFormat || 'mm/dd/yy'

            // Use jQuery datepicker if available, otherwise fallback to native date input
            if ($.fn.datepicker) {
                input.datepicker({
                    dateFormat: dateFormat,
                    minDate: field.minDate || null,
                    maxDate: field.maxDate || null,
                    changeMonth: true,
                    changeYear: true,
                    yearRange: field.yearRange || '-100:+10',
                    showButtonPanel: true
                })
                trigger.on('click', () => {
                    if (!input.prop('disabled')) input.datepicker('show')
                })
            } else {
                // Fallback: convert to native date input
                input.prop('readonly', false)
                input.prop('type', 'date')
                trigger.on('click', () => {
                    if (!input.prop('disabled')) input.trigger('click')
                })
            }
        }
    }

    static { Autoform.fieldInitializers.date = Autoform.fieldInitializers.calendar }

    // ─── Constructor ────────────────────────────────────────────────

    /**
     * @param {AutoformConfig} attrs - Form configuration
     */
    constructor(attrs = {}) {
        const secondaryAction = attrs.secondaryAction || 'reset'
        const defaultSecondaryLabel = secondaryAction === 'skip' ? 'Skip' : 'Reset'

        const params = {
            template: Autoform.template,
            autoInit: false,
            submitLabel: attrs.submitLabel || 'Submit',
            secondaryLabel: attrs.secondaryLabel || defaultSecondaryLabel,
            ...attrs
        }
        super(params)

        this.fields = attrs.fields || []
        this.secondaryAction = secondaryAction
        this.showSecondary = attrs.showSecondary !== false
        this._lastData = null
        this._lastStringified = null

        I18n.loadTranslations('en', {
            youSkipped: 'You skipped the form.'
        })
    }

    /**
     * Shallow-merge two config objects. `map1` properties win over `map2`.
     * @param {Object} map1 - High-priority overrides
     * @param {Object} map2 - Base defaults
     * @returns {Object} Merged config
     */
    static merge(map1, map2) {
        const merged = {}
        for (const key in map2) merged[key] = map2[key]
        for (const key in map1) merged[key] = map1[key]
        return merged
    }

    // ─── Lifecycle ──────────────────────────────────────────────────

    /**
     * Load CSS, render all fields, run initializers, and wire up buttons.
     * Call manually since `autoInit` is `false`.
     * @returns {Promise<void>}
     */
    async init() {
        Widget.loadCss('../css/forms/autoform.css', import.meta.url)
        if (!this.showSecondary) this.find('#secondary').hide()
        this._renderFields()
        this._initializeFields()
        this._attachEventHandlers()
    }

    /**
     * Widget lifecycle hook — caches form data before the DOM is removed
     * so that {@link getData} and {@link stringify} still return values
     * after the form is destroyed.
     */
    beforeDestroy() {
        // Cache data before DOM is removed
        if (!this._lastData) {
            try { this._lastData = this._extractData() } catch (e) { /* DOM may already be gone */ }
        }
        if (!this._lastStringified) {
            try { this._lastStringified = this._buildStringify() } catch (e) { /* DOM may already be gone */ }
        }
    }

    // ─── Overridable Hooks ──────────────────────────────────────────

    /**
     * Called after validation passes, just before the form is destroyed.
     * Override in subclasses to perform side-effects or cancel submission.
     * @param {Object<string,*>} data - Validated field data
     * @returns {false|void} Return `false` to prevent submit and keep the form open
     */
    onBeforeSubmit(data) {
        // Override in subclasses
    }

    /**
     * Called when validation fails. Default implementation highlights invalid fields.
     * Override to provide custom error display (e.g. toast notifications).
     * @param {ValidationError[]} errors
     */
    onValidationError(errors) {
        this._showFieldErrors(errors)
    }

    // ─── Rendering ──────────────────────────────────────────────────

    /**
     * Iterate {@link fields}, resolve each template, and append to the `<form>` element.
     * @private
     */
    _renderFields() {
        const formEl = this.find('form')
        this.fields.forEach(field => {
            const tmpl = Autoform.templates[field.type]
            if (!tmpl) {
                console.warn(`Autoform: Unknown field type "${field.type}" for field "${field.name}"`)
                return
            }
            const values = this._resolveFieldValues(field)
            const html = tmpl.trim().replace(/\{\{([^}]+)\}\}/g, (match, key) => {
                return values[key] !== undefined ? values[key] : ''
            })
            formEl.append(html)
        })
    }

    /**
     * Build the `{{key}}` → value map used to resolve a field's template.
     * Handles I18n label resolution, HTML attribute flags, and JSON serialization for multi-types.
     * @param {FieldDef} field
     * @returns {Object<string, string|number>}
     * @private
     */
    _resolveFieldValues(field) {
        const label = field.label ? I18n.t(field.label) : field.name
        const values = {
            name: field.name,
            label: label,
            value: field.value || field.defaultValue || '',
            placeholder: field.placeholder || label,
            min: field.min ?? 0,
            max: field.max ?? 100,
            step: field.step ?? 1,
            rows: field.rows ?? 3,
            required: field.required ? 'required="required"' : '',
            disabled: field.disabled ? 'disabled' : '',
            checked: field.checked ? 'checked' : '',
            multiple: field.multiple ? 'multiple="multiple"' : '',
            size: field.size ? `size="${field.size}"` : '',
            options: this._renderOptions(field)
        }

        // For multi-types, ensure value is a JSON array string
        if (field.type === 'multiDropdown' || field.type === 'multiValue') {
            const arr = Array.isArray(field.value) ? field.value : []
            values.value = JSON.stringify(arr)
        }

        return values
    }

    /**
     * Render `<option>` elements for select/dropDown/multiDropdown fields.
     * Accepts both string arrays and `{value, label}` object arrays.
     * @param {FieldDef} field
     * @returns {string} HTML string of `<option>` elements
     * @private
     */
    _renderOptions(field) {
        const optionsList = field.options || field.enum
        if (!Array.isArray(optionsList) || optionsList.length === 0) {
            return '<option value="" disabled selected>No options available</option>'
        }
        let html = '<option value="" disabled selected>Select...</option>'
        optionsList.forEach(option => {
            const val = typeof option === 'string' ? option : option.value
            const label = typeof option === 'string' ? option : option.label
            const selected = val === field.defaultValue ? ' selected' : ''
            html += `<option value="${val}"${selected}>${label}</option>`
        })
        return html
    }

    /**
     * Run post-render initializers for complex field types (range, multiDropdown, etc.).
     * @private
     */
    _initializeFields() {
        this.fields.forEach(field => {
            const initializer = Autoform.fieldInitializers[field.type]
            if (initializer) initializer(this, field)
        })
    }

    // ─── Event Handlers ─────────────────────────────────────────────

    /**
     * Wire up submit and secondary (reset/skip) button click handlers.
     * Submit flow: clear errors → validate → cache data → onBeforeSubmit → emit → destroy.
     * @private
     */
    _attachEventHandlers() {
        this.find('#submit').on('click', () => {
            this._clearFieldErrors()

            const errors = this.validate()
            if (errors.length > 0) {
                this.onValidationError(errors)
                return
            }

            // Cache data before destroying DOM
            this._lastData = this._extractData()
            this._lastStringified = this._buildStringify()

            const result = this.onBeforeSubmit(this._lastData)
            if (result === false) return

            this.emit(Autoform.SUBMIT_EVENT, {
                data: this._lastData,
                stringified: this._lastStringified
            })

            this.destroy()
        })

        this.find('#secondary').on('click', () => {
            if (this.secondaryAction === 'skip') {
                this._lastStringified = I18n.t('youSkipped')
                this.emit(Autoform.SKIP_EVENT, {
                    stringified: this._lastStringified
                })
                this.destroy()
            } else {
                this.reset()
            }
        })
    }

    /**
     * Reset all fields to their initial values (from the field definitions).
     * Multi-value types re-run their initializers to rebuild tag UI.
     * Emits {@link Autoform.RESET_EVENT}.
     */
    reset() {
        this._clearFieldErrors()
        this.fields.forEach(field => {
            const name = field.name
            switch (field.type) {
                case 'checkbox':
                    this.find(`#${name}`).prop('checked', !!field.checked)
                    break
                case 'rating':
                    this.find(`input[name="${name}"]`).prop('checked', false)
                    break
                case 'multiDropdown':
                case 'multiValue': {
                    const arr = Array.isArray(field.value) ? field.value : []
                    this.find(`#${name}`).val(JSON.stringify(arr))
                    // Re-run initializer to rebuild tags
                    const initializer = Autoform.fieldInitializers[field.type]
                    if (initializer) initializer(this, field)
                    break
                }
                case 'range': {
                    const val = field.value || field.defaultValue || field.min || 0
                    this.find(`#${name}`).val(val)
                    this.find(`#${name}-label`).text(val)
                    break
                }
                default:
                    this.find(`#${name}`).val(field.value || field.defaultValue || '')
            }
        })
        this.emit(Autoform.RESET_EVENT)
    }

    // ─── Validation ─────────────────────────────────────────────────

    /**
     * Validate all fields against their constraints (required, pattern, min/max, custom).
     * @returns {ValidationError[]} Empty array if valid
     */
    validate() {
        const errors = []
        const allData = this._extractData()

        this.fields.forEach(field => {
            const value = allData[field.name]
            const fieldErrors = this._validateField(field, value, allData)
            if (fieldErrors.length > 0) {
                errors.push({
                    field: field.name,
                    label: field.label ? I18n.t(field.label) : field.name,
                    errors: fieldErrors
                })
            }
        })

        return errors
    }

    /**
     * Validate a single field value against its definition constraints.
     * @param {FieldDef} field
     * @param {*} value - Current field value
     * @param {Object<string,*>} allData - All field values (for cross-field validators)
     * @returns {string[]} Error messages (empty if valid)
     * @private
     */
    _validateField(field, value, allData) {
        const errors = []
        const isEmpty = value === null || value === undefined || value === '' ||
            (Array.isArray(value) && value.length === 0)

        if (field.required && isEmpty) {
            errors.push('This field is required')
            return errors
        }

        if (isEmpty) return errors

        if (field.pattern) {
            const regex = new RegExp(field.pattern)
            if (!regex.test(String(value))) {
                errors.push(field.patternMessage || 'Invalid format')
            }
        }

        if (field.minLength && String(value).length < field.minLength) {
            errors.push(`Minimum ${field.minLength} characters`)
        }

        if (field.maxLength && String(value).length > field.maxLength) {
            errors.push(`Maximum ${field.maxLength} characters`)
        }

        if (field.min !== undefined && (field.type === 'number' || field.type === 'range')) {
            if (Number(value) < field.min) errors.push(`Minimum value is ${field.min}`)
        }

        if (field.max !== undefined && (field.type === 'number' || field.type === 'range')) {
            if (Number(value) > field.max) errors.push(`Maximum value is ${field.max}`)
        }

        if (field.customValidator) {
            const result = field.customValidator(value, field, allData)
            if (result) errors.push(result)
        }

        return errors
    }

    /**
     * Add `.has-error` class and error message divs to invalid fields.
     * @param {ValidationError[]} errors
     * @private
     */
    _showFieldErrors(errors) {
        errors.forEach(({ field, errors: msgs }) => {
            const el = this.find(`#${field}`).closest('.autoform-field')
            el.addClass('has-error')
            msgs.forEach(msg => {
                el.append(`<div class="autoform-error-message">${msg}</div>`)
            })
        })
    }

    /**
     * Remove all validation error indicators and messages from the form.
     * @private
     */
    _clearFieldErrors() {
        this.find('.autoform-field').removeClass('has-error')
        this.find('.autoform-error-message').remove()
    }

    // ─── Data Extraction ────────────────────────────────────────────

    /**
     * Get all field values as a plain object.
     * Safe to call after destroy — returns the cached snapshot.
     * @returns {Object<string,*>} Field name → value map
     */
    getData() {
        if (this._destroyed || !this.node) return this._lastData || {}
        return this._extractData()
    }

    /**
     * Read current values from the live DOM for every field.
     * Handles type-specific extraction (checkbox → boolean, multi → array, file → FileList).
     * @returns {Object<string,*>}
     * @private
     */
    _extractData() {
        const data = {}
        this.fields.forEach(field => {
            const name = field.name
            switch (field.type) {
                case 'rating':
                    const checked = this.find(`input[name="${name}"]:checked`)
                    data[name] = checked.length > 0 ? checked.val() : null
                    break
                case 'checkbox':
                    data[name] = this.find(`#${name}`).prop('checked')
                    break
                case 'multiDropdown':
                case 'multiValue':
                    try { data[name] = JSON.parse(this.find(`#${name}`).val()) || [] }
                    catch { data[name] = [] }
                    break
                case 'file':
                    data[name] = this.find(`#${name}`)[0]?.files || null
                    break
                default:
                    data[name] = this.find(`#${name}`).val()
            }
        })
        return data
    }

    /**
     * Build a `FormData` object suitable for multipart/form-data uploads.
     * File fields are appended as `File` objects; arrays are JSON-stringified.
     * @returns {FormData}
     */
    getFormData() {
        const data = this._extractData()
        const formData = new FormData()
        this.fields.forEach(field => {
            if (field.type === 'file') {
                const files = this.find(`#${field.name}`)[0]?.files
                if (files && files.length > 0) {
                    for (let i = 0; i < files.length; i++) formData.append(field.name, files[i])
                }
            } else {
                const val = data[field.name]
                formData.append(field.name, Array.isArray(val) ? JSON.stringify(val) : val ?? '')
            }
        })
        return formData
    }

    /**
     * Human-readable HTML summary of all field values.
     * Safe to call after destroy — returns the cached snapshot.
     * @returns {string} HTML string with `<p>` elements per field
     */
    stringify() {
        if (this._destroyed || !this.node) return this._lastStringified || ''
        return this._buildStringify()
    }

    /**
     * Generate the HTML summary from live DOM data.
     * @returns {string}
     * @private
     */
    _buildStringify() {
        const data = this._extractData()
        let s = ''
        this.fields.forEach(field => {
            const label = field.label ? I18n.t(field.label) : field.name
            const value = data[field.name]
            if (field.type === 'checkbox') {
                s += `<p>${label}: ${value ? 'Yes' : 'No'}</p>\n`
            } else if (Array.isArray(value)) {
                s += `<p>${label}: ${value.join(', ')}</p>\n`
            } else {
                s += `<p>${label}: ${value || ''}</p>\n`
            }
        })
        return s
    }
